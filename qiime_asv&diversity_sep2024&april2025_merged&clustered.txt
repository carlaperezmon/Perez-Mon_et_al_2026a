tables_apr=('/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_april_2025/16S/150_150/table-dada2_16S.qza' '/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_april_2025/18S/170_150/table-dada2_18S.qza'

seqs_sep=('/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_september_2024/16S/170_150/merged_A5J3/rep-seqs-dada2_16S.qza' '/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_september_2024/18S/170_150/rep-seqs-dada2_18S.qza')
seqs_apr=('/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_april_2025/16S/150_150/rep-seqs-dada2_16S.qza' '/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_april_2025/18S/170_150/rep-seqs-dada2_18S.qza')

mkdir 16S 18S

for ((i=0; i<${#marker_names[@]}; i++)) 
do

cd ${marker_names[i]}

qiime feature-table merge \
    --i-tables   ${tables_sep[i]}  ${tables_apr[i]} \
    --o-merged-table table-dada2_merged_${marker_names[i]}.qza

qiime feature-table merge-seqs \
  --i-data  ${seqs_sep[i]}  ${seqs_apr[i]}\
  --o-merged-data rep-seqs-dada2_merged_${marker_names[i]}.qza

cd $merged_dir

done 

#somehow the above merging didnt work so I had to do it manually

qiime feature-table merge \
    --i-tables  '/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_september_2024/16S/170_150/merged_A5J3/table-dada2_16S.qza'  '/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_april_2025/16S/170_150/table-dada2_16S.qza' \

qiime feature-table merge-seqs \
  --i-data  '/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_september_2024/16S/170_150/merged_A5J3/rep-seqs-dada2_16S.qza' '/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_april_2025/16S/170_150/rep-seqs-dada2_16S.qza' \
  --o-merged-data rep-seqs-dada2_merged_${marker_names[i]}.qza

cd $merged_dir

i=1

cd ${marker_names[i]}

qiime feature-table merge \
    --i-tables  '/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_september_2024/18S/170_150/table-dada2_18S.qza'  '/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_april_2025/18S/170_150/table-dada2_18S.qza' \
    --o-merged-table table-dada2_merged_${marker_names[i]}.qza

qiime feature-table merge-seqs \
  --i-data  '/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_september_2024/18S/170_150/rep-seqs-dada2_18S.qza' '/exports/cmvm/eddie/eb/groups/ogden_grp/carla/sequencing_data/data_analyses/who_april_2025/18S/170_150/rep-seqs-dada2_18S.qza' \
  --o-merged-data rep-seqs-dada2_merged_${marker_names[i]}.qza

cd $merged_dir




##### cluster the sequences to reduce seq batch effects ######

for ((i=0; i<${#marker_names[@]}; i++)) #to loop through the length of array directories (same for tablets below). helped by chatgpt :). 
do

cd ${marker_names[i]}

qiime vsearch cluster-features-de-novo \
--i-sequences rep-seqs-dada2_merged_${marker_names[i]}.qza \
--i-table table-dada2_merged_${marker_names[i]}.qza \
--p-perc-identity 0.99 \
--o-clustered-sequences rep-seqs_clustered_${marker_names[i]}.qza \
--o-clustered-table table_clustered_${marker_names[i]}.qza \
--verbose

cd $merged_dir

done


#check ordination in clustered sequences

sampling_depth=(5000 5000)

for ((i=0; i<${#marker_names[@]}; i++)) #to loop through the length of array directories (same for tablets below). helped by chatgpt :). 
do
	cd ${marker_names[i]}
	pwd

qiime phylogeny align-to-tree-mafft-fasttree \
  --i-sequences rep-seqs_clustered_${marker_names[i]}.qza \
  --o-alignment aligned-rep-seqs-dada2.qza \
  --o-masked-alignment masked-aligned-rep-seqs-dada2.qza \
  --o-tree unrooted-tree-dada2.qza \
  --o-rooted-tree rooted-tree-dada2.qza

qiime diversity core-metrics-phylogenetic \
  --i-phylogeny rooted-tree-dada2.qza \
  --i-table table_clustered_${marker_names[i]}.qza \
  --p-sampling-depth ${sampling_depth[i]} \
  --m-metadata-file $metadata_file \
  --p-ignore-missing-samples \
  --output-dir core-metrics-results_clustered_sdepth${sampling_depth[i]}
 
cd $merged_dir

done 



###### Species annotation #######

#for merged

cd $merged_dir

for ((i=0; i<${#marker_names2[@]}; i++)) #to loop through the length of array directories (same for tablets below). helped by chatgpt :).

#run for Silva
qiime feature-classifier classify-sklearn \
  --i-classifier ${classifier[i]} \
  --i-reads rep-seqs-dada2_merged_${marker_names[i]}.qza \

#run for gg and pr2
qiime feature-classifier classify-sklearn \
  --i-classifier ${classifier2[i]} \
  --i-reads rep-seqs-dada2_merged_${marker_names[i]}.qza \

#for clustered

for ((i=0; i<${#marker_names[@]}; i++)) #to loop through the length of array directories (same for tablets below). helped by chatgpt :). 
do

cd ${marker_names[i]}

#run for Silva
qiime feature-classifier classify-sklearn \
  --i-classifier ${classifier[i]} \
  --i-reads rep-seqs_clustered_${marker_names[i]}.qza \

#run for gg and pr2
qiime feature-classifier classify-sklearn \
  --i-classifier ${classifier2[i]} \
  --i-reads rep-seqs_clustered_${marker_names[i]}.qza \

done
